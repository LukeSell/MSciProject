52  auto t = s;
53 
54  // Apply move to solution
55  t.order[m.indexA] = s.order[m.indexB];
56  t.order[m.indexB] = s.order[m.indexA];
57  t.moveMade = m;
58 
59  // Calculate updated cost of solution efficiently
60 
61  // Handle left edge of B
62  t.cost -= distanceMatrix[m.cityA][s.order[m.indexB - 1].name];
63  t.cost += distanceMatrix[m.cityB][t.order[m.indexB - 1].name];
64 
65  // Handle right edge of B, with potential for last element
66  if (m.indexB != sz - 1) {
67  t.cost -= distanceMatrix[m.cityA][s.order[m.indexB + 1].name];
68  t.cost += distanceMatrix[m.cityB][t.order[m.indexB + 1].name];
69  } else {
70  t.cost -= distanceMatrix[m.cityA][s.order[0].name];
71  t.cost += distanceMatrix[m.cityB][t.order[0].name];
72  }
73 
74  // Handle left edge of A, with potential for first element
75  if (m.indexA != 0) {
76  t.cost -= distanceMatrix[m.cityB][s.order[m.indexA - 1].name];
77  t.cost += distanceMatrix[m.cityA][t.order[m.indexA - 1].name];
78  } else {
79  t.cost -= distanceMatrix[m.cityB][s.order[s.order.size() - 1].name];
